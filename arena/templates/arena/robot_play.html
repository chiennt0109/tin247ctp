// C++ gợi ý
#include <bits/stdc++.h>
using namespace std;
int main() {
    char c;
    if (!(cin >> c)) return 0;
    if (c == 'U') cout << "UP";
    else if (c == 'D') cout << "DOWN";
    else if (c == 'L') cout << "LEFT";
    else if (c == 'R') cout << "RIGHT";
    return 0;
}
