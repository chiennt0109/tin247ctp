using namespace std;
int n,m;
vector<vector<int>> adj;
vector<int> visited;

void dfs(int u){
    visited[u]=1;
    cout<<u<<" ";
    for(int v:adj[u])
        if(!visited[v]) dfs(v);
}

void bfs(int start){
    queue<int> q;
    q.push(start);
    visited[start]=1;
    while(!q.empty()){
        int u=q.front();q.pop();
        cout<<u<<" ";
        for(int v:adj[u])
            if(!visited[v]){
                visited[v]=1;
                q.push(v);
            }
    }
}

    }

    cout<<"\\nBFS: ";
    visited.assign(n+1,0);
    bfs(1);
}

    adj[u].append(v)

    vis[u]=True
    print(u,end=' ')
    for v in adj[u]:
        if not vis[v]:
            dfs(v,vis)

    while q:

        print(u,end=' ')
        for v in adj[u]:
            if not vis[v]:
                vis[v]=True
                q.append(v)
