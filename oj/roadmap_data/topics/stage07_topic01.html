using namespace std;

int n;
vector<int> cur;
vector<int> used;

void backtrack() {
    if ((int)cur.size() == n) {
        for (int x : cur) cout << x << " ";
        cout << "\n";
        return;
    }
    for (int val = 1; val <= n; val++) {
        if (!used[val]) {
            used[val] = 1;
            cur.push_back(val);
            backtrack();
            cur.pop_back();
            used[val] = 0;
        }
    }
}

int main() {
    cin >> n;
    used.assign(n+1, 0);
    backtrack();
}
