    int n,W;cin>>n>>W;
    vector<int> w(n+1),v(n+1);
    for(int i=1;i<=n;i++)cin>>w[i]>>v[i];
    vector<vector<int>> dp(n+1,vector<int>(W+1,0));
    for(int i=1;i<=n;i++)
        for(int j=0;j<=W;j++)

    cout<<dp[n][W];
