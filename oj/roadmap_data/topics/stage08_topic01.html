#include <bits/stdc++.h>
using namespace std;
int main(){
    stack<int> st;
    queue<int> q;
    deque<int> dq;
    for(int i=1;i<=3;i++){
        st.push(i);
        q.push(i);
        dq.push_back(i);
    }
    cout<<"Stack top: "<<st.top()<<"\\n";
    cout<<"Queue front: "<<q.front()<<"\\n";
    cout<<"Deque back: "<<dq.back()<<"\\n";
}

from collections import deque
st = []
q = deque()
dq = deque()

for i in range(1,4):
    st.append(i)
    q.append(i)
    dq.append(i)

print("Stack top:", st[-1])
print("Queue front:", q[0])
print("Deque back:", dq[-1])
